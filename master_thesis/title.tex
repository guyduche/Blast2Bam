\makeatletter
\def\clap#1{\hbox to 0pt{\hss #1\hss}}
\def\haut#1#2#3{
    \hbox to \hsize{
        \rlap{\vtop{\raggedright #1}}
        \hss
        \clap{\vtop{\centering #2}}
        \hss
        \llap{\vtop{\raggedleft #3}}}}

\def\middle#1#2#3{
    \hbox to \hsize{
        \rlap{\vtop{\raggedright #1}}
        \hss
        \clap{\vtop{\centering #2}}
        \hss
        \llap{\vtop{\raggedleft #3}}}}

\def\bas#1#2#3{
    \hbox to \hsize{
        \rlap{\vbox{\raggedright #1}}
        \hss
        \clap{\vbox{\centering #2}}
        \hss
        \llap{\vbox{\raggedleft #3}}}}

\def\maketitle{
    \thispagestyle{empty}\vbox to \vsize{
        \haut{\@institut}{\@blurb}{\@univ}
        \vfill
        \vspace{1cm}
        \begin{center}
            \usefont{T1}{ptm}{m}{n}
            \Huge \@title
        \end{center}
        \vspace{1ex}
        \hrule height 4pt
        \vspace{1ex}
        \middle{\usefont{T1}{ptm}{m}{n}\Large \@author}{}{supervisor \usefont{T1}{ptm}{m}{n}\Large \@supervisor}
        \vfill
        \vfill
        \vbox{\centering \@location, \@date}
        \vspace{1cm}
        \bas{}{\@cnrs \hsp \@inserm}{}}}

\def\date#1{\def\@date{#1}}
\def\author#1{\def\@author{#1}}
\def\supervisor#1{\def\@supervisor{#1}}
\def\title#1{\def\@title{#1}}
\def\location#1{\def\@location{#1}}
\def\blurb#1{\def\@blurb{#1}}
\def\institut#1{\def\@institut{#1}}
\def\univ#1{\def\@univ{#1}}
\def\cnrs#1{\def\@cnrs{#1}}
\def\inserm#1{\def\@inserm{#1}}
\makeatother
